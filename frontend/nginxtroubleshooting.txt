# configuration file /etc/nginx/nginx.conf:
user www-data;
worker_processes auto;
pid /run/nginx.pid;
include /etc/nginx/modules-enabled/*.conf;

events {
    worker_connections 768;
}

http {

    sendfile on;
    tcp_nopush on;
    types_hash_max_size 2048;
    server_tokens off;

    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent"';

    access_log /var/log/nginx/access.log main;
    error_log  /var/log/nginx/error.log;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;

    limit_req_zone $binary_remote_addr zone=slowbots:10m rate=5r/s;

    map $http_user_agent $bad_bot {
        default             0;
        "~*GPTBot"          1;
        "~*ChatGPT-User"    1;
        "~*ClaudeBot"       1;
        "~*Claude-Web"      1;
        "~*anthropic"       1;
        "~*CCBot"           1;
        "~*Google-Extended" 1;
        "~*FacebookBot"     1;
        "~*Bytespider"      1;
        "~*PerplexityBot"   1;
        "~*Amazonbot"       1;
        "~*cohere-ai"       1;
        "~*Diffbot"         1;
        "~*AhrefsBot"       1;
        "~*SemrushBot"      1;
        "~*MJ12bot"         1;
        "~*DotBot"          1;
        "~*PetalBot"        1;
        "~*python-requests" 1;
        "~*wget"            1;
        "~*Go-http-client"  1;
        ""                  1;
    }

    # Catch-all — drop unknown hosts
    server {
        listen 80 default_server;
        listen 443 ssl default_server;
        server_name _;
        ssl_certificate     /etc/nginx/ssl/dummy.crt;
        ssl_certificate_key /etc/nginx/ssl/dummy.key;
        return 444;
    }

    # HTTP → HTTPS redirect
    server {
        listen 80;
        server_name journal.williamthomas.name;
        return 301 https://$host$request_uri;
    }

    # Main HTTPS server
    server {
        listen 443 ssl;
        server_name journal.williamthomas.name;

        ssl_certificate     /etc/letsencrypt/live/journal.williamthomas.name/fullchain.pem;
        ssl_certificate_key /etc/letsencrypt/live/journal.williamthomas.name/privkey.pem;

        if ($bad_bot) { return 403; }

        limit_req zone=slowbots burst=20 nodelay;

        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header X-XSS-Protection "1; mode=block" always;
        add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

        # Proxy everything to FastAPI (serves both API and React SPA)
        location / {
            proxy_pass http://127.0.0.1:8000;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_read_timeout 120s;
        }

        access_log /var/log/nginx/access.log main;
        error_log  /var/log/nginx/error.log;
    }

    include /etc/nginx/conf.d/*.conf;
}
